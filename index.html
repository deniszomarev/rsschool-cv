var moveZeros = function (arr) {
    let removedItemsCount = 0;
    for (var i in arr) {
        if (arr[i-removedItemsCount] === 0) {
            arr.splice(i-removedItemsCount, 1);
            arr = arr.concat ([0]);
            removedItemsCount++;
        }
    }
    return arr;
    };
